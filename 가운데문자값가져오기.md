  public static String solution(String s) {
    String answer = "";
      String[] t = s.split("");
      if(s.length()%2 == 0) {
          answer = t[s.length()/2-1] + t[s.length()/2];
      }else {
          answer = t[s.length()/2];
      }
      return answer;
  }
